class CreditRiskEngine:
    def predict_default(self, company_data, method='ensemble'):
        if method == 'cnn_lstm':
            model = ModelFactory.create(ModelType.CNN_LSTM_CREDIT)
            return model.predict_default(company_data)
        elif method == 'ensemble':
